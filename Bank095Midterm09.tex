\newproblem{09a}
{
	Solve the equation.
	
	$6(2x-3)+8=12x-(2x+2)$
}
{
	\begin{tabular}{l r}
	$12x-18+8=12x-2x-2$ & 3 pts to here\\
	$12x-10=10x-2$ & 4 pts to here\\
	$2x-10=-2$ & 5 pts to here\\
	$2x=8$ & 6 pts to here\\
	$x=4$ & 7 pts to here
	\end{tabular}
}
